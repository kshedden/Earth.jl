        dp[:, v] .= mean(D[:, v])
    end
    qq = quantile(D[:, zvar], [0.2, 0.5, 0.8])
    for j in 1:3
        i = 50*(j-1)
        dp[(i+1):(i+50), xvar] = range(extrema(D[:, xvar])..., 50)
        dp[(i+1):(i+50), zvar] .= qq[j]
    end
    yh = predict(md, dp)
    return dp[:, xvar], yh, qq
end;
